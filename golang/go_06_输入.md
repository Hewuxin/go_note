让用户输入数据，完成项目交互。

- fmt.Scan

  ```go
  // 示例1
  var name string
  fmt.Println("请输入用户名：")
  fmt.Scan(&name)
  fmt.Printf(name)
  
  // 示例2
  var name string
  var age int
  fmt.Println("请输入用户名：")
  // 会返回两个值：count，用户输入了几个值；err用户输入错误则是错误信息
  _, err := fmt.Scan(&name, &age)
  if err == nil {
  	fmt.Printf(name, age)    
  }else{
      fmt.Println("输入有误", err)
  }
  //特别说明:fmt.Scan 要求输入两个值，必须输入两个值，否则他会一直等待。
  ```

- fmt.Scanln

  ```go
  // 示例1
  var name string
  fmt.Print("请输入用户名：")
  fmt.Scanln(&name)
  fmt.Printf(name)
  
  var name string
  var age int
  fmt.Print("请输入用户名：")
  // 会返回两个值：count，用户输入了几个值；err用户输入错误则是错误信息
  count, err := fmt.Scanln(&name, &age)
  fmt.Println(count,err)
  fmt.Println(name,age)
  if err == nil {
      fmt.Println(name,age)
  }else{
      fmt.Println("输入错误", err)
  }
  // 特别说明：fmt.Scanln 等待回车，遇到回车就结束
  ```

- fmt.Scanf

  ```go
  var name string
  var age int
  fmt.Print("请输入用户名")
  fmt.Scanf("我叫%s 今年%d 岁 ", &name, &age)
  _, _ = fmt.Scanf("我叫%s 今年%d 岁 ", &name, &age)
  fmt.Println(name, age)
  //
  ```

获取一行里面含空格的信息

```go
reader := bufio.NewReader(os.Stdin)
//line 从stdin中读取一行的数据（字节集合 -> 转换成为字符串）
// reader 默认一次能读取4096个字节（4096/3个汉字）
//    1. 一次性读完， isPerfix = false
//    2. 先读一部分， isperfix = True,再去读取，如果读完isPerfix = false
line, _, _ := reader.ReadLine()
data := string(line)
fmt.Println(data)
```
